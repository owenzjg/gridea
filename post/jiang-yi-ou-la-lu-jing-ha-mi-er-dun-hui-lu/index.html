一. 欧拉路径

欧拉路径是指在图中找到一条路径，使得这条路径经过图中的每条边仅一次。
欧拉路径存在的条件：图中的奇点只有两个或者零个。
（奇点：与当前点相连的边数目有奇数个的点）
如果有奇点，则奇点为起点或终点。
当欧拉路径的起点和终点相同时，这是一条欧拉回路。
欧拉回路存在的条件：图中的奇点是零个。

二. 哈密尔顿回路

哈密尔顿回路是指在图中存在一条回路，使得每个点都被经过且经过一次。

欧拉回路是指不重复地走过所有路径的回路，而哈密尔顿环是指不重复地走过所有的点，并且最后还能回到起点的回路

三. 算法模板

欧拉路径

#include <iostream>
using namespace std;
int n, m, g[101][101], d[101], cnt, ans[101];
void dfs(int x) {
    for(int i=1; i<=n; i++)
        if(g[x][i]) {
            g[x][i] = g[i][x] = 0;
            dfs(i);
        }
    ans[++cnt] = x;
}
int main() {
    cin >> n >> m;
    int u, v;
    for(int i=1; i<=m; i++) {
        cin >> u >> v;
        g[u][v] = g[v][u] = 1;
        d[u] ++;
        d[v] ++;
    }
    int st = 1;
    for(int i=1; i<=n; i++)
        if(d[i]%2==1) {
            st = i;
            break;
        }
    dfs(st);
    for(int i=cnt; i>=1; i--)
        cout << ans[i] << ' ';
    return 0;
}

哈密尔顿路径

#include <iostream>
using namespace std;
int n, m, g[105][105], ans[105];
bool vis[105];
bool dfs(int x, int t) {
    if(t>n) {
        if(g[x][1])    return true;
        return false;
    }
    for(int i=1; i<=n; i++) {
        if(g[x][i] && !vis[i]) {
            ans[t] = i;
            vis[i] = true;
            if(dfs(i, t+1))    return true;
            vis[i] = false;
        }
    }
    return false;
}
int main() {
    cin >> n >> m;
    int u, v;
    for(int i=1; i<=m; i++) {
        cin >> u >> v;
        g[u][v] = g[v][u] = 1;
    }
    ans[1] = 1;
    vis[1] = true;
    if(dfs(1, 2)) {
        for(int i=1; i<=n; i++)
            cout << ans[i] << ' ';
        cout << ans[1];
    }
    else    cout << "NO";
    return 0;
}
