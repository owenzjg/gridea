[c++]二叉树便利模板（前中后便利）

整理了一下，顺便分享一波，为二叉树的前序遍历，中序遍历和后序遍历，需要那个就保留那个函数，注释非常详细

//一个二叉树的各种便利方式
#include <bits/stdc++.h>
using namespace std;
#define maxn 10000//更改为题目中的n数据即可
struct node{
    int l,r;//分别代表左孩子和右孩子
}tree[maxn+10];
void qian(int a){//前序遍历
    cout<<a<<" ";//输出当前节点
    if(tree[a].l!=-1) qian(tree[a].l);//如果左孩子存在，访问左孩子
    if(tree[a].r!=-1) qian(tree[a].r);//如果右孩子存在，访问右孩子
    //整体的就是访问当前，左孩子，右孩子，为前序遍历
    return ;//结束
}
void zhong(int a){//中序遍历
    if(tree[a].l!=-1) zhong(tree[a].l);//如果左孩子存在，访问左孩子
    cout<<a<<" ";//输出当前节点
    if(tree[a].r!=-1) zhong(tree[a].r);//如果右孩子存在，访问右孩子
    //整体的就是访问左孩子，当前，右孩子，为中序遍历
    return ;//结束
}
void hou(int a){//后序遍历
    if(tree[a].l!=-1) hou(tree[a].l);//如果左孩子存在，访问左孩子
    if(tree[a].r!=-1) hou(tree[a].r);//如果右孩子存在，访问右孩子
    cout<<a<<" ";//输出当前节点
    //整体的就是访问左孩子，右孩子，当前为中序遍历
    return ;//结束
}
int main(){
    int n;
    cin>>n;//节点总数
    for(int i=1;i<=n;i++){
        int a,b;
        cin>>a>>b;//题目输入描述：第i行的两个数为i的左孩子和右孩子
        tree[i].l=a,tree[i].r=b;//录入链式存储
    }
    qian(1);
    cout<<endl;
    zhong(1);
    cout<<endl;
    hou(1);
    cout<<endl;
    return 0;
}

原理的话就是一个递归，进入后分别访问孩子和根节点，以来模仿前中后访问输出结果。
这是一个比较常见的板子，还是需要熟练掌握的。
