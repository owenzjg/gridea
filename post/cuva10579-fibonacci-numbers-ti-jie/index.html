题目意思

输入n，输入第n个Fibonacci数（即斐波那契数列）

题目范围指出输入的斐波那契数不会超过1000位。

题目思路

斐波那契数自古就是考高精度的好理由

这里需要用到c++的高精度加法，其实可以理解为一个比较大的模拟，用数组和循环 按照人计算加法时的思路一步一步的去计算。

代码

#include <bits/stdc++.h>
using namespace std;
int a[10010][1010];
int main() {
    a[1][1] = 1;
    a[2][1] = 1;
    for (int i = 3; i <= 10000; i++) {
        for (int j = 1; j <= 1000; j++) {
            a[i][j] += a[i - 1][j] + a[i - 2][j];
            if (a[i][j] > 9) {
                a[i][j + 1] += a[i][j] / 10;
                a[i][j] %= 10;
            }
        }
    }
    int x;
    while (cin >> x) {
        bool flag = 0;
        for (int i = 1000; i >= 1; i--) {
            if (a[x][i] != 0 || flag) {
                flag = 1;
                cout << a[x][i];
            }
        }
        cout << endl;
    }
    return 0;
}
