小C的数学[math]

这道题目看起来非常简单，但是比较容易被坑。

第一道题目一定要细心，不然就凉透了

首先看最大的数据范围。

b,cb,cb,c 都到 1018 ， 直接上 long long,而且肯定不可以for便利。

正解的话有很多种，这里我就直接上我的解法了，还是比较好理解的。

    long long a, b, c;
    cin >> a >> b >> c;
    b = ceil((b * 1.0 / a));
    c = floor((c * 1.0 / a));
    cout << c - b + 1;
    return 0;

当然，还有很多别的方法，比如说两边是的就 +1+1+1 等

    long long a, b, c;
    cin >> a >> b >> c;
    if(c%a!=0 && b%a!=0){
        cout<<(c-b)/a<<endl;
    }else{
        cout<<(c-b)/a+1<<endl;
    }
    return 0;

小C的数组[array]

首先，写双重 for 循环肯定超时，是下下策。

而且，这道题目要求求和，必须要注意。

106 * 106 = 1012

肯定要开long long。

这个题目就是找一个规律，也是有很多种做法。

ll n;
cin >> n;
ll cnt = 0;
ll last = 0;
for (ll i = 1; i <= n; i++) {
    cin >> a[i];
    if (a[i] != a[i - 1])
        last = i - 1;
    t[a[i]] = i;
    cnt += t[a[last]];
}
cout << cnt;

或者说，用个 fff 数组递推一下也可以。

long long he=0;
cin>>n;
for(int i=1;i<=n;i++){
    cin>>a[i];
    if(a[i]!=a[i-1]){
        f[i]=i-1;
    }else{
        f[i]=f[i-1];
    }
    he+=f[i];
}
cout<<he;

小C的数 [number]

这道题目实在不会可以拿部分分。

题目一定要认真读，不要读错题目了。

先瞅一眼骗分写法：

  1. dfs -- 40分 -- O(2n)
    对于每一个数，选或不选。选的话就还要把前面所有的便利判断一遍。
    最后全部选完过后打擂台。
  2. 最长上升子序列 -- 60分 -- O(n2)
    这个题目虽然说要取子序列，但是最后只要取长度，所以说选择的顺序并不重要。
    核心代码如下：
    sort(a, a + n);
 int maxx = 0;
 for (int i = 0; i < n; i++) {
     for (int j = 0; j < i; j++) {
         if (a[i] % a[j] == 0) {
             f[i] = max(f[i], f[j]);
         }
     }
     f[i]++;
     maxx = max(maxx, f[i]);
 }
 cout << maxx;

最终的正解类是一个dp，还是比较好理解的。时间限时是2秒，正好擦边球过的

int n;
scanf("%d", &n);
int maxx = 0;
int ans = 0;
for (int i = 1; i <= n; i++) {
    int x;
    scanf("%d", &x);
    cnt[x]++;
    f[x]++;
    ans = max(ans, f[x]);
    maxx = max(maxx, x);
}
for (int i = maxx; i >= 1; i--) {
    for (int k = 2; k * i <= maxx; k++) {
        f[i] = max(f[i], f[i * k] + cnt[i]);
        ans = max(ans, f[i]);
    }
}
printf("%d", ans);

小C的计算 [sum]

没啥好讲的，先写一个质数判断的函数

bool zhi(int x) {
    if (x < 2)
        return 0;
    for (int i = 2; i * i <= x; i++) {
        if (x % i == 0)
            return 0;
    }
    return 1;
}

然后 for 便利即可

int x, y;
cin >> x >> y;
long long cnt = 0;
for (int i = x; i <= y; i++) {
    if (zhi(i))
        cnt += i;
}
cout << cnt;

小C的工作 [work]

这个题目不太方便文字叙述，两张图片即可

看完这个就比较好理解了，再自己那一张草稿纸哗啦哗啦就可以了。

ll n;
cin>>n;
ll i;
for(i=1;i<=n;i++){
	cin>>a[i][0]>>a[i][1];
	x=max(x,a[i][0]+a[i][1]);
	y=min(a[i][0],y);
}
for(i=1;i<=n;i++){
	x-=a[i][1];
}
cout<<min(x,y);

当然，这个如果不是特别可以理解的话也可以用另一种稍微更费时间的算法.
先定义一下node类型和cmp排序函数

struct node {
    long long begin, end, len;
} a[200010];
bool cmp(node x, node y) {
    return x.end < y.end;
}

然后main函数非常好理解

int n;
int i;
cin >> n;
for (i = 0; i < n; i++) {
    cin >> a[i].begin >> a[i].len;
    a[i].end = a[i].begin + a[i].len;
}
sort(a, a + n, cmp);
long long ans = a[n - 1].begin;
for (i = n - 2; i >= 0; i--) {
    if (ans <= a[i].end) {
        ans -= a[i].len;
    } else {
        ans = a[i].begin;
    }
}
cout << ans;
